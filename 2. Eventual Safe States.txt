class Solution {
    private:
    bool checkcycle(int node, int vis[], int pathvis[], int check[], vector<int> adj[])
    {
        vis[node] = pathvis[node] = 1;
        check[node] = 0;
        for(auto it: adj[node])
        {
            if(!vis[it])
            {
                if(checkcycle(it, vis, pathvis, check, adj))
                {
                    check[it] = 0;
                    return true;
                }
            }
            else if(pathvis[it])
            {
                check[it] = 0;
                return true;
            }
        }
        check[node] = 1;
        pathvis[node] = 0;
        return false;
    }
  public:
    vector<int> eventualSafeNodes(int V, vector<int> adj[]) {
        int vis[V] = {0};
        int pathvis[V] = {0};
        int check[V] = {0};
        vector<int> ans;
        for(int i = 0; i < V; i++)
        {
            if(!vis[i])
            {
                checkcycle(i, vis, pathvis, check, adj);
            }
        }
        for(int i = 0; i < V; i++)
        {
            if(check[i]) ans.push_back(i);
        }
        return ans;
    }
};

int main() {
    int t;
    cin >> t;
    while (t--) {

        int V, E;
        cin >> V >> E;
        vector<int> adj[V];

        for (int i = 0; i < E; i++) {
            int u, v;
            cin >> u >> v;
            adj[u].push_back(v);
        }

        Solution obj;
        vector<int> safeNodes = obj.eventualSafeNodes(V, adj);
        for (auto i : safeNodes) {
            cout << i << " ";
        }
        cout << endl;
    }
}
