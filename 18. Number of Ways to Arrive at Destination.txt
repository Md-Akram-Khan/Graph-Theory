class Solution {
  public:
    int countPaths(int n, vector<vector<int>>& roads) {
        vector<pair<int,int>> adj[n];
        for(auto it : roads)
        {
            adj[it[0]].push_back({it[1], it[2]});
            adj[it[1]].push_back({it[0], it[2]});
        }
        priority_queue<pair<long long int,int>, vector<pair<long long int,int>>, greater<pair<long long int, int>>> pq;
        pq.push({0, 0});
        int mod = (int)(1e9 + 7);
        int INF = (int)(1e18);
        vector<long long int> dis(n, INF), ways(n, 0);
        dis[0] = 0;
        ways[0] = 1;
        while(!pq.empty())
        {
            long long int d = pq.top().first;
            int parent = pq.top().second;
            pq.pop();
            for(auto it : adj[parent])
            {
                int child = it.first;
                int cost = it.second;
                long long int dd = d + cost;
                if(dd < dis[child])
                {
                    dis[child] = dd;
                    pq.push({dis[child], child});
                    ways[child] = ways[parent]; 
                }
                else if(dd == dis[child])
                {
                    ways[child] = (ways[child]+ways[parent]) % mod;
                }
            }
        }
        return ways[n-1] % mod;
    }
};
